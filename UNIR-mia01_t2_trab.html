# define la funcin mcd
def mcd(a, b):
    resto = 0
    while(b > 0):
        resto = b
        b = a % b
        a = resto
    return a

# define la funcin gcd (recursiva) y mcm
def gcd(x, y):
    return x if not y else gcd(y, x % y)
 
def mcm(x, y):
    return x * y / gcd(x, y)

def elimina_tildes(cadena):
    a,b = '','aeiouu'
    trans = str.maketrans(a,b)
    return cadena.translate(trans)    
    return cadena

  1. Lee el fichero proporcionado.

  3. Cuenta el nmero de veces que aparece el artculo el en el fichero (nota: elemento contiene los caracteres el pero no debe contar puesto que no contiene a el artculo el.
  4. Elimina las tildes del fichero (elimina_tildes) y convierta todas las letras a minscula (linea_lower)

# abre fichero texto, modalidad lectura
f = open("texto.txt", "r")

# define variables de trabajo. COntador de llineas y un array "nueva_linea" para escribir en el nuevo texto.
contador_linea = 0 # inicializa en 0
ocurrencias = 0 # inicializa en 0
nueva_linea = []

# lee el texto linea a linea 
for linea in f:
        
    #contador de lineas del texto
    contador_linea = contador_linea + 1
           
    #conversor linea a minusculas. 
    linea_lower = linea.lower()
    
    #elimina tildes
    linea_sin_tildes = elimina_tildes(linea_lower)
    
    #print(linea_sin_tildes)
    
    nueva_linea.append(linea_sin_tildes)
    
    #COntador de cocurrencias de articulo "el" ya en minusculas y sin tildes
    ocurrencias = ocurrencias + linea.count("el ")
    
    
f.close()
